算法

通用部分


function exch(a, i, j) {
	var temp = a[i];
	a[i] = a[j];
	a[j] = temp;
}

冒泡排序


function bubbleSort(a) {
	var len = a.length;
	if (len == 0)
		return [];

	for (var i = 0; i < len -1; i++) {
		for (var j = i; j < len; j++) {
			if (a[j] > a[j+1])
				exch(a, j, j+1);
		}
	}
	return a;
}

桶排序

function bucketSort(a) {
	var len = a.length;
	if (len == 0) {
		return []
	}

	var res = [];
	var bucket = [];
	var i = 0,
		j = 0,
		min = 0;
		max = 0;

	for (i = 0; i < len ; i++) {
		if (min > a[i])
			min = a[i];
		if (max < a[i])
			max = a[i];
	}

	//count of buckets  53 9
	var count = Math.ceil((max - min) / 10);

	for (i = 0; i < len; i++) {
		var current = a[i];
		for (j = 0; j < count; j++) {
			if (current >= min + j*10 && current < min + (j+1)*10) {
				if (!bucket[j])
					bucket[j] = [];
				bucket[j].push(current);
				bucket[j] = insertSort(bucket[j]);
				//  对每个桶里的元素进行插入排序
			}
		}
	}

	for (i = 0; i < count; i++) {
		for (j = 0; j < bucket[i].length; j++) {
			res.push(bucket[i][j]);
		}
	}

    return res;

}

计数排序

function countSort(a) {
	var len = a.length;
	if (len == 0)
		return [];

	var min = a[0];
	var max = a[0];
	var bucket = [];
	var res = [];

	for (let i = 1; i < len; i++) {
		if (min > a[i]) {
			min = a[i];
		}
		if (max < a[i]) {
			max = a[i];
		}
	}

	for (let i = 0; i < len; i++) {
		let index = a[i] - min;  // 为了防止第一个数很大的时候浪费空间，所以每次都从桶的0开始
		if (bucket[index] == undefined) {
			bucket[index] = 1;
		}
		else {
			bucket[index]++;
		}
	}

	for (let i = 0; i < max-min+1; i++) {
		if (bucket[i] != undefined) {
			let count = bucket[i]
			while (count > 0) {
				res.push(i+min);
				count--;
			}
		}
	}

	return res;
}

插入排序

function insertSort(a) {
	var len = a.length;
	if (len == 0)
		return [];

	var i = 0,
		current = 0;

	for (i = 1; i < len; i++) {
		current = i;
		while (current >= 1 && a[current] < a[current-1]) {
			exch(a, current, current-1);
		}
	}

	return a;
}

选择排序

function selectSort(a) {
	var len = a.length;
	if (len == 0) {
		return [];
	}
	var minValue = a[0];
	var minIndex = 0;
	var i = 0,
		j = 0;

	for (i = 0; i < a.length-1; i++) {
		minValue = a[i];
		minIndex = i;
		for (j = i+1; j < a.length; j++) {
			if (minValue > a[j]) {
				minIndex = j;
				minValue = a[j];
			}
		}
		exch(a, i, minIndex);
	}
	return a;
}

堆排序（数组第一个位置用不用分出两种方法）

//  数组第一个位置留空
function heapSort(a) {
	let N = a.length;

	for (let k = Math.floor(N/2); k>= 1; k--) {
		sink(a, k, N);
	}

	while(N > 1) {
		exch(a, 1, N--);
		sink(a, 1, N);
	}

}

function sink(a, k, N) {
	while (2 * k <= N) {
		var j = 2 * k;
		if (j < N && a[j]<a[j+1])
			j++;

		if (a[k] > a[j])
			break;
		exch(a, k, j);
		k = j;
	}
}

// 第一个位置不留空
function heapSort(a) {
	let N = a.length - 1;

	for (let k = Math.ceil((N-1)/2); k>= 0; k--) {
		sink(a, k, N);
	}

	while(N > 0) {
		exch(a, 0, N--);
		sink(a, 0, N);
	}
	console.log(a);

}

function sink(a, k, N) {
	while (2 * k + 1 <= N) {
		var j = 2 * k + 1;
		if (j < N && a[j]<a[j+1])
			j++;

		if (a[k] > a[j])
			break;
		exch(a, k, j);
		k = j;
	}
}

基数排序

function radixSort(a) {
	var len = a.length;
	if (len == 0) {
		return [];
	}

	var res = [];
	var bucket = [];
	var isBreak = false;
	var count = 1;

	while (!isBreak) {
		isBreak = true;
		bucket = [];  // 注意每次桶都要更新..因为这里我被困了好久
		for (var i = 0; i < len; i++) {
			var remain = Math.floor(a[i] / count) % 10;
			if (remain > 9)
				remain = 0;
			if (remain != 0) {
				isBreak = false;
			}
			if (!bucket[remain]) {
				bucket[remain] = [];
			}
			bucket[remain].push(a[i]);
		}
		a = [];   // 和桶一样每次要更新
		for (var j = 0; j < bucket.length; j++) {
			if (Array.isArray(bucket[j]) ) {
				for (var k = 0; k < bucket[j].length; k++) {
					a.push(bucket[j][k]);
				}
			}
		}
		count *= 10;
	}

	return a;
}
